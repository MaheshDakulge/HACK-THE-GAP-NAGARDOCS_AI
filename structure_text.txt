1. PRODUCT REQUIREMENTS DOCUMENT 
1.1  Problem Statement (SG2) 
Municipal corporations across India manage millions of physical government documents — marksheets, birth 
certificates, property records, income certificates, ration cards — entirely through manual paper-based processes. 
This results in slow retrieval, data loss, no searchability, and zero analytics. NagarDocs AI solves this with one 
scan. 
1.2  Solution — What NagarDocs AI Does 
• 
Teacher / staff scans or uploads any government document (PDF or image) 
• 
EasyOCR extracts raw text from the document 
• 
ChatGPT API classifies document type and extracts structured fields automatically 
• 
Document saved to searchable digital cabinet (like File Explorer) 
• 
Staff can search any record instantly by name, ID, or document type 
• 
Analytics dashboard shows total records, types, languages, tamper alerts 
• 
Export to Excel for any document category in one click

ocr_service.py 
# Uses EasyOCR to extract text from PDF / image 
# Input:  raw file bytes 
# Output: raw text string 
# Supports: English, Hindi, Marathi 
# pip install easyocr 
 
import easyocr 
reader = easyocr.Reader(['en', 'hi'])  # load once at startup 
 
async def extract_text(file_bytes: bytes) -> str: 
    result = reader.readtext(file_bytes, detail=0) 

    return ' '.join(result) 
 
ai_service.py  (THE SECRET WEAPON) 
# ChatGPT classifies ANY document + extracts fields + detects tamper 
# Input:  raw OCR text 
# Output: structured JSON dict 
 
async def classify_and_extract(ocr_text: str) -> dict: 
    response = openai.chat.completions.create( 
        model='gpt-4o-mini', 
        messages=[{ 
            'role': 'user', 
            'content': f''' 
Extract structured data from this Indian government document. 
Return ONLY JSON: 
{ 
  document_type: marksheet|birth_certificate|property_record| 
                 ration_card|income_certificate|other, 
  confidence_score: 0-100, 
  language: english|hindi|marathi|mixed, 
  extracted_fields: {name, date, id_number, issuing_authority, other_fields}, 
  is_tampered: true|false, 
  tamper_reason: string or null, 
  summary: one line 
} 
Document: {ocr_text}''' 
        }], 
        response_format={'type': 'json_object'}, 
        temperature=0 
    ) 
    return json.loads(response.choices[0].message.content) 
 
cabinet_service.py 
# Auto-sort: reads all unorganized docs → creates folders → assigns docs 
 
async def auto_sort(supabase): 
    docs = supabase.table('documents').select('*').is_('cabinet_id', None).execute() 
    groups = {} 
    for doc in docs.data: 
        dtype = doc['document_type'] 
        groups.setdefault(dtype, []).append(doc['id']) 
    for dtype, ids in groups.items(): 
        cab = supabase.table('cabinets').upsert( 
            {'name': dtype.replace('_',' ').title()+'s', 'type': 'auto'}, 
            on_conflict='name').execute() 
        cab_id = cab.data[0]['id'] 
        supabase.table('documents').update({'cabinet_id': cab_id}).in_('id', 
ids).execute()

4.2  New Tables SQL 
documents table 
CREATE TABLE documents ( 
  id               UUID PRIMARY KEY DEFAULT uuid_generate_v4(), 
  cabinet_id       UUID REFERENCES cabinets(id) ON DELETE SET NULL, 
  document_type    TEXT,       -- marksheet|birth_certificate|property_record|... 
  confidence       INT,        -- 0-100 from ChatGPT 
  language         TEXT,       -- english|hindi|marathi|mixed 
  extracted_fields JSONB,      -- {name, date, id_number, issuing_authority, other} 
  is_tampered      BOOLEAN DEFAULT FALSE, 
  tamper_reason    TEXT, 
  summary          TEXT,       -- one-line GPT summary 
  raw_text         TEXT,       -- full OCR output 
  filename         TEXT, 
  file_url         TEXT,       -- Supabase Storage URL 
  created_at       TIMESTAMPTZ DEFAULT NOW() 
); 
 
-- Full text search index 
CREATE INDEX idx_docs_search ON documents 
  USING GIN(to_tsvector('english', raw_text)); 
CREATE INDEX idx_docs_type ON documents(document_type); 
CREATE INDEX idx_docs_cabinet ON documents(cabinet_id); 
 
cabinets table 
CREATE TABLE cabinets ( 

  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(), 
  name        TEXT NOT NULL UNIQUE,  -- 'Marksheets', 'Birth Certificates' 
  type        TEXT DEFAULT 'manual', -- auto | manual 
  color       TEXT DEFAULT '#1A3C5E', 
  icon        TEXT DEFAULT 'folder', 
  created_at  TIMESTAMPTZ DEFAULT NOW() 
); 
 
Existing tables — no changes needed 
teachers, projects, upload_jobs, students, 
student_results, subject_marks, templates 
-- all remain exactly as before
